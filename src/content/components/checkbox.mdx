## Import

```js
import { Checkbox } from "@prismane/core";
```

## Usage

```jsx
function Demo() {
  const [value, setValue] = useState("");

  return (
    <Checkbox
      name="agreement"
      label="I agree that Prismane is the best!"
      value={value}
      onChange={(e) => setValue(e.target.checked)}
    />
  );
}
```
