## Import

```js
import { Toaster } from "@prismane/core";
```

## Usage

```jsx
function Demo() {
  const Inner = () => {
    const toast = useToast();

    return (
      <Button
        onClick={() => {
          toast({
            element: <Alert>Hello, World!</Alert>,
          });
        }}
      >
        Click to show toast
      </Button>
    );
  };

  return (
    <Toaster>
      <Inner />
    </Toaster>
  );
}
```

<Important>Always wrap the application with the `Toaster` component!</Important>
