## Import

```js
import { Flex } from "@prismane/core";
```

## Usage

```jsx
<Flex gap={fr(2)} h={100} w="100%">
  <Flex bg={["base", 700]} w={120}>
    1
  </Flex>
  <Flex bg={["base", 700]} w={120}>
    2
  </Flex>
</Flex>
```

```jsx
<Flex gap={fr(2)} h={100} w="100%">
  <Flex bg={["base", 700]} w={120}>
    1
  </Flex>
  <Flex bg={["base", 700]} grow>
    2
  </Flex>
</Flex>
```

```jsx
<Flex gap={fr(2)} h={100} w="100%" justify="between">
  <Flex bg={["base", 700]} w={120}>
    1
  </Flex>
  <Flex bg={["base", 700]} w={120}>
    2
  </Flex>
</Flex>
```

### Versatile Component

```jsx
<Flex as="strong" gap={fr(5)}>
  <Box>Hello</Box>
  <Box>World</Box>
</Flex>
```
