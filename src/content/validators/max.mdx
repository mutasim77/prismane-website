# max Validator

The `max` validator checks if a value is longer than the given length, if it is true, it returns an error message, if not it returns `null`.

## Basic Usage

```tsx
import { max } from "@prismane/core/validators";

console.log(max("Ivan is very cool", 100, "Username")); // Will return null
console.log(max("Ivan is very cool", 2, "Username"));
// Will return `Username has to be shorter than 100 characters!`
```

## useForm Hook Usage

```tsx
import { Form, TextField, Button } from "@prismane/core";
import { useForm } from "@prismane/core/hooks";

const Complex = () => {
  const { handleSubmit, handleReset, register } = useForm({
    fields: {
      username: {
        value: "",
        validators: {
          max: (v: string) => max(v, 4, "Username"),
        },
      },
    },
  });

  return (
    <Form
      onSubmit={(e: any) => {
        handleSubmit(e, (v: any) => console.log(v));
      }}
      onReset={() => handleReset()}
      maw={300}
    >
      <TextField
        placeholder="Enter username: "
        label="Username:"
        {...register("username")}
      />
      <Button type="submit">Submit</Button>
    </Form>
  );
};
```
