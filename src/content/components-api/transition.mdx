<Versatile>Transition</Versatile>

Transition display a container that can transition it's CSS properties.

## Import

```js
import { Transition } from "@prismane/core";
```

## Usage

By default, `Transition` transitions all CSS properties.

```jsx
function Demo() {
  const [toggled, setToggled] = useState(false);

  return (
    <Flex direction="column" gap={fr(5)}>
      <Button onClick={() => setToggled(!toggled)}>Toggle Properties</Button>
      <Transition
        w={fr(30)}
        h={fr(30)}
        br={toggled ? "base" : "lg"}
        bg={toggled ? "base" : "primary"}
      />
    </Flex>
  );
}
```

### Transition Properties

`Transition` has the following properties:

- `all` - all,
- `colors` - color, background-color, border-color, text-decoration-color, fill, stroke,
- `opacity` - opacity,
- `shadow` - box-shadow, filter,
- `transform` - transform,

```jsx
function Demo() {
  const [toggled, setToggled] = useState(false);

  return (
    <Flex gap={fr(5)}>
      <Flex direction="column" gap={fr(5)}>
        <Button onClick={() => setToggled(!toggled)}>Toggle Properties</Button>
        <Flex gap={fr(5)}>
          <Transition
            transition="all"
            w={fr(30)}
            h={fr(30)}
            br={toggled ? "base" : "lg"}
            bg={toggled ? "base" : "primary"}
            op={toggled ? 0.5 : 1}
            bsh={toggled ? "sm" : "lg"}
          />
          <Transition
            transition="colors"
            w={fr(30)}
            h={fr(30)}
            br={toggled ? "base" : "lg"}
            bg={toggled ? "base" : "primary"}
            op={toggled ? 0.5 : 1}
            bsh={toggled ? "sm" : "lg"}
          />
          <Transition
            transition="opacity"
            w={fr(30)}
            h={fr(30)}
            br={toggled ? "base" : "lg"}
            bg={toggled ? "base" : "primary"}
            op={toggled ? 0.5 : 1}
            bsh={toggled ? "sm" : "lg"}
          />
          <Transition
            transition="shadow"
            w={fr(30)}
            h={fr(30)}
            br={toggled ? "base" : "lg"}
            bg={toggled ? "base" : "primary"}
            op={toggled ? 0.5 : 1}
            bsh={toggled ? "sm" : "lg"}
          />
          <Transition
            transition="transform"
            w={fr(30)}
            h={fr(30)}
            br={toggled ? "base" : "lg"}
            bg={toggled ? "base" : "primary"}
            op={toggled ? 0.5 : 1}
            bsh={toggled ? "sm" : "lg"}
          />
        </Flex>
      </Flex>
    </Flex>
  );
}
```
